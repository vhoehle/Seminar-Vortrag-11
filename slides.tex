\begin{frame}{1. Rekursionen}

\textbf{Idee:}
Rekursionen ermöglichen es, neue Werte aus bereits bekannten kleineren Werten zu berechnen.

\vspace{0.4cm}

Zur Erinnerung betrachten wir die bereits bekannten
\textbf{Binomialkoeffizienten}:

\[
\binom{n}{k}
=
\frac{n!}{k!(n-k)!}
\]

\[
\binom{n}{k}
=
\binom{n}{n-k}
\]

\[
\binom{n}{k}
=
\frac{n(n-1)\cdots(n-k+1)}{k!}
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Erweiterung der Binomialkoeffizienten auf \(\mathbb{Z}\) und \(\mathbb{C}\) }

\begin{itemize}
    
    \item Zunächst setzen wir
    \[
    \binom{0}{0}=1,
    \]
    da die leere Menge genau eine Teilmenge besitzt, nämlich sich selbst.

    \item Außerdem setzen wir
    \[
    0! = 1
    \]
    sowie
    \[
    n^{\underline{0}} = n^{\overline{0}} = 1.
    \]

    \item Für die Fakultät benötigen wir zunächst \(k \geq 0\).
\end{itemize}

\vspace{0.2cm}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Definition 1.1}

\vspace{0.2cm}

Für \(r\in\mathbb{C}\) und \(k\in\mathbb{Z}\) definieren wir

\[
\binom{r}{k}
=
\begin{cases}
\dfrac{r(r-1)\cdots(r-k+1)}{k!}, & k\geq 0,\\[1ex]
0, & k<0.
\end{cases}
\]

\vspace{0.4cm}

Die abfallende Fakultät wird definiert durch

\[
r^{\underline{k}}
=
r(r-1)\cdots(r-k+1).
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.2 (Pascalsche Identität)}

\vspace{0.3cm}

Für \(r\in\mathbb{C}\) und \(k\in\mathbb{Z}\) gilt

\[
\binom{r}{k}
=
\binom{r-1}{k-1}
+
\binom{r-1}{k}.
\]

\vspace{0.5cm}

\textbf{Beweis:} Mittels der Polynommethode.

\begin{itemize}
    \item Für \(k<0\) gilt unmittelbar \(0=0\).

    \item Für \(k=0\) gilt unmittelbar \(1=1\).

    \item Sei nun \(k\geq 1\).
\end{itemize}

Wir betrachten die Identität

\[
\binom{x}{k}
\stackrel{?}{=}
\binom{x-1}{k-1}
+
\binom{x-1}{k},
\qquad x\in\mathbb{C}.
\]

Beide Seiten sind Polynome in \(x\) vom Grad \(k\).

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.2 (Fortsetzung)}

Die beiden Polynome stimmen für alle
\(x\in\mathbb{N}\) überein.

Da zwei Polynome vom Grad höchstens \(k\),
die an unendlich vielen Stellen übereinstimmen,
identisch sind, folgt

\[
\binom{x}{k}
=
\binom{x-1}{k-1}
+
\binom{x-1}{k}.
\]

für alle \(x\in\mathbb{C}\).

Insbesondere gilt dies für \(x=r\).

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.3 (Pascalsches Dreieck)}

\vspace{0.3cm}

Aus der Pascalschen Identität ergibt sich für
\(n,k\in\mathbb{N}_0\) das Pascalsche Dreieck:

\[
\begin{array}{c|cccccc}
n\backslash k & 0 & 1 & 2 & 3 & 4 & 5\\
\hline
0 & 1\\
1 & 1 & 1\\
2 & 1 & 2 & 1\\
3 & 1 & 3 & 3 & 1\\
4 & 1 & 4 & 6 & 4 & 1\\
5 & 1 & 5 & 10 & 10 & 5 & 1
\end{array}
\]

\vspace{0.2cm}

Jeder Eintrag ergibt sich als Summe der beiden darüberliegenden Einträge.

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.4 (Zeilensumme)}

\vspace{0.3cm}

Für \(n\in\mathbb N_0\) gilt

\[
\sum_{k=0}^{n}\binom{n}{k}=2^n.
\]

\vspace{0.3cm}

\textbf{Beweisidee:}

Vollständige Induktion über \(n\).

Unter Verwendung der Pascalschen Identität erhält man

\[
\sum_{k=0}^{n+1}\binom{n+1}{k}
=
2\sum_{k=0}^{n}\binom{n}{k}.
\]

Mit der Induktionsvoraussetzung folgt

\[
2\cdot 2^n = 2^{n+1}.
\]

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.5 (Spaltensumme)}

\vspace{0.3cm}

Für \(n,k\ge 0\) gilt

\[
\sum_{m=0}^{n}\binom{m}{k}
=
\binom{n+1}{k+1}.
\]

\vspace{0.5cm}

\textbf{Korollar 1.6 (Diagonalsumme)}

\vspace{0.3cm}

Für \(m,n\ge 0\) gilt

\[
\sum_{k=0}^{n}\binom{m+k}{k}
=
\binom{m+n+1}{n}.
\]

\vspace{0.4cm}

\textit{Die Beweise erfolgen analog durch vollständige Induktion und sind in der Ausarbeitung enthalten.}

\end{frame}
