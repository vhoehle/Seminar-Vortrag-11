\begin{frame}{1. Rekursionen}

\textbf{Idee:}
Rekursionen ermöglichen es, neue Werte aus bereits bekannten kleineren Werten zu berechnen.

\vspace{0.4cm}

Zur Erinnerung betrachten wir die bereits bekannten
\textbf{Binomialkoeffizienten}:

\[
\binom{n}{k}
=
\frac{n!}{k!(n-k)!}
\]

\[
\binom{n}{k}
=
\binom{n}{n-k}
\]

\[
\binom{n}{k}
=
\frac{n(n-1)\cdots(n-k+1)}{k!}
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Erweiterung der Binomialkoeffizienten auf \(\mathbb{Z}\) und \(\mathbb{C}\) }

\begin{itemize}
    
    \item Zunächst setzen wir
    \[
    \binom{0}{0}=1,
    \]
    da die leere Menge genau eine Teilmenge besitzt, nämlich sich selbst.

    \item Außerdem setzen wir
    \[
    0! = 1
    \]
    sowie
    \[
    n^{\underline{0}} = n^{\overline{0}} = 1.
    \]

    \item Für die Fakultät benötigen wir zunächst \(k \geq 0\).
\end{itemize}

\vspace{0.2cm}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Definition 1.1}

\vspace{0.2cm}

Für \(r\in\mathbb{C}\) und \(k\in\mathbb{Z}\) definieren wir

\[
\binom{r}{k}
=
\begin{cases}
\dfrac{r(r-1)\cdots(r-k+1)}{k!}, & k\geq 0,\\[1ex]
0, & k<0.
\end{cases}
\]

\vspace{0.4cm}

Die abfallende Fakultät wird definiert durch

\[
r^{\underline{k}}
=
r(r-1)\cdots(r-k+1).
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.2 (Pascalsche Identität)}

\vspace{0.3cm}

Für \(r\in\mathbb{C}\) und \(k\in\mathbb{Z}\) gilt

\[
\binom{r}{k}
=
\binom{r-1}{k-1}
+
\binom{r-1}{k}.
\]

\vspace{0.5cm}

\textbf{Beweis:} Mittels der Polynommethode.

\begin{itemize}
    \item Für \(k<0\) gilt unmittelbar \(0=0\).

    \item Für \(k=0\) gilt unmittelbar \(1=1\).

    \item Sei nun \(k\geq 1\).
\end{itemize}

Wir betrachten die Identität

\[
\binom{x}{k}
\stackrel{?}{=}
\binom{x-1}{k-1}
+
\binom{x-1}{k},
\qquad x\in\mathbb{C}.
\]

Beide Seiten sind Polynome in \(x\) vom Grad \(k\).

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.2 (Fortsetzung)}

Die beiden Polynome stimmen für alle
\(x\in\mathbb{N}\) überein.

Da zwei Polynome vom Grad höchstens \(k\),
die an unendlich vielen Stellen übereinstimmen,
identisch sind, folgt

\[
\binom{x}{k}
=
\binom{x-1}{k-1}
+
\binom{x-1}{k}.
\]

für alle \(x\in\mathbb{C}\).

Insbesondere gilt dies für \(x=r\).

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.3 (Pascalsches Dreieck)}

\vspace{0.3cm}

Aus der Pascalschen Identität ergibt sich für
\(n,k\in\mathbb{N}_0\) das Pascalsche Dreieck:

\[
\begin{array}{c|cccccc}
n\backslash k & 0 & 1 & 2 & 3 & 4 & 5\\
\hline
0 & 1\\
1 & 1 & 1\\
2 & 1 & 2 & 1\\
3 & 1 & 3 & 3 & 1\\
4 & 1 & 4 & 6 & 4 & 1\\
5 & 1 & 5 & 10 & 10 & 5 & 1
\end{array}
\]

\vspace{0.2cm}

Jeder Eintrag ergibt sich als Summe der beiden darüberliegenden Einträge.

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.4 (Zeilensumme)}

\vspace{0.3cm}

Für \(n\in\mathbb N_0\) gilt

\[
\sum_{k=0}^{n}\binom{n}{k}=2^n.
\]

\vspace{0.5cm}

\textbf{Beweis.}

Per vollständiger Induktion über \(n\).

\[
\text{I.A.:}\qquad
\sum_{k=0}^{0}\binom{0}{k}
=
\binom{0}{0}
=
1
=
2^0.
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Korollar 1.4 (Fortsetzung)}

\vspace{0.3cm}

\[
\text{I.V.:}\qquad
\exists n\in\mathbb N_0:
\sum_{k=0}^{n}\binom{n}{k}
=
2^n.
\]

\vspace{0.3cm}

\[
\text{I.S.:}\qquad
\text{Zu zeigen: }
\sum_{k=0}^{n+1}\binom{n+1}{k}
=
2^{n+1}.
\]

Mit der Pascalschen Identität folgt

\[
\sum_{k=0}^{n+1}\binom{n+1}{k}
=
2\sum_{k=0}^{n}\binom{n}{k}.
\]

Mit der Induktionsvoraussetzung ergibt sich

\[
2\sum_{k=0}^{n}\binom{n}{k}
=
2\cdot 2^n
=
2^{n+1}.
\]

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.5 (Spaltensumme)}

\vspace{0.3cm}

Für \(n,k\ge 0\) gilt

\[
\sum_{m=0}^{n}\binom{m}{k}
=
\binom{n+1}{k+1}.
\]

\vspace{0.2cm}

\textit{Beweis siehe Ausarbeitung.}

\vspace{0.5cm}

\textbf{Korollar 1.6 (Diagonalsumme)}

\vspace{0.3cm}

Für \(m,n\ge 0\) gilt

\[
\sum_{k=0}^{n}\binom{m+k}{k}
=
\binom{m+n+1}{n}.
\]

\vspace{0.2cm}

\textit{Beweis s. Ausarbeitung.}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.7 (Negation)}

\vspace{0.3cm}

Für \(r\in\mathbb C\) und \(k\in\mathbb Z\) gilt

\[
\binom{-r}{k}
=
(-1)^k
\binom{r+k-1}{k}.
\]

\vspace{0.4cm}

\textbf{Beweis.}

\[
(-x)^{\underline{k}}
=
(-x)(-x-1)\cdots(-x-k+1)
\]

\[
=
(-1)^k\,x(x+1)\cdots(x+k-1)
\]

\[
=
(-1)^k\,x^{\overline{k}}.
\]

(Reziprozitätsgesetz)

\vspace{0.2cm}

Durch Division durch \(k!\) folgt die Behauptung.

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Korollar 1.8 (Alternierende Zeilensumme)}

\vspace{0.3cm}

Für \(m,n\in\mathbb N_0\) gilt

\[
\sum_{k=0}^{m}
(-1)^k
\binom{n}{k}
=
(-1)^m
\binom{n-1}{m}.
\]

\vspace{0.5cm}

Insbesondere folgt für \(m=n\)

\[
\sum_{k=0}^{n}
(-1)^k
\binom{n}{k}
=
0.
\]

\vspace{0.5cm}

\textit{Beweis s. Ausarbeitung.}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.9 (Binomialsatz)}

\vspace{0.3cm}

Für \(n\geq 0\) gilt

\[
(x+y)^n
=
\sum_{k=0}^{n}
\binom{n}{k}
x^k y^{\,n-k}.
\]

\vspace{0.5cm}

Insbesondere folgt für \(y=1\)

\[
(x+1)^n
=
\sum_{k=0}^{n}
\binom{n}{k}
x^k.
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.9}

\vspace{0.3cm}

Schreiben wir

\[
(x+y)^n
=
\underbrace{(x+y)\cdots(x+y)}_{n\text{-mal}}.
\]

Beim Ausmultiplizieren wählen wir aus jedem Faktor
entweder \(x\) oder \(y\).

\vspace{0.3cm}

Ein Term der Form

\[
x^k y^{\,n-k}
\]

entsteht genau dann, wenn aus genau \(k\) der
\(n\) Faktoren ein \(x\) gewählt wird und aus den
restlichen \(n-k\) Faktoren ein \(y\).

\vspace{0.3cm}

Die Anzahl dieser Möglichkeiten beträgt

\[
\binom{n}{k}.
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.9 (Fortsetzung)}

\vspace{0.3cm}

Somit erscheint der Term

\[
x^k y^{\,n-k}
\]

mit dem Koeffizienten

\[
\binom{n}{k}.
\]

\vspace{0.4cm}

Summiert man über alle möglichen Werte

\[
k=0,\dots,n,
\]

so erhält man

\[
(x+y)^n
=
\sum_{k=0}^{n}
\binom{n}{k}
x^k y^{\,n-k}.
\]

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.10 (Vandermonde-Identität)}

\vspace{0.2cm}

Für \(r,s,n\in\mathbb N_0\) gilt

\[
\binom{r+s}{n}
=
\sum_{k=0}^{n}
\binom{r}{k}
\binom{s}{n-k}.
\]

\vspace{0.3cm}

\textbf{Beweis.}

Zunächst betrachten wir den Fall
\(r,s\in\mathbb N_0\).

Seien \(R\) und \(S\) disjunkte Mengen mit

\[
|R|=r,\qquad |S|=s.
\]

Wir zählen die Anzahl der Möglichkeiten,
\(n\) Elemente aus \(R\cup S\) auszuwählen.

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.10 (Fortsetzung)}

\begin{enumerate}
\item Direkte Auswahl von \(n\) Elementen aus \(R\cup S\):

\[
\binom{r+s}{n}
\]

Möglichkeiten.

\vspace{0.2cm}

\item Sei \(k\) die Anzahl der ausgewählten Elemente aus \(R\).

Dann stammen \(n-k\) Elemente aus \(S\).

Aus \(R\): \(\binom{r}{k}\) Möglichkeiten,

aus \(S\): \(\binom{s}{n-k}\) Möglichkeiten.

\end{enumerate}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.10 (Fortsetzung)}

Nach der Produktregel gibt es für festes \(k\)

\[
\binom{r}{k}\binom{s}{n-k}
\]

Möglichkeiten.

Summation über alle \(k=0,\ldots,n\) ergibt

\[
\sum_{k=0}^{n}
\binom{r}{k}\binom{s}{n-k}.
\]

Da beide Zählweisen dieselbe Menge erfassen, folgt

\[
\binom{r+s}{n}
=
\sum_{k=0}^{n}
\binom{r}{k}\binom{s}{n-k}.
\]

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Definition 1.11 (Stirling-Zahlen erster Art)}

\vspace{0.3cm}

Die Stirling-Zahl erster Art

\[
s_{n,k}
\]

gibt die Anzahl der Permutationen einer
\(n\)-elementigen Menge an, die genau
\(k\) Zyklen besitzen.

\vspace{0.4cm}

Es gelten die Anfangswerte

\[
s_{0,0}=1,
\qquad
s_{0,k}=0 \ (k>0),
\qquad
s_{n,0}=0 \ (n>0).
\]

\vspace{0.4cm}

\textbf{Beispiel:}

\[
s_{4,3}=6.
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.12 (Stirling-Zahlen erster Art)}

\vspace{0.2cm}

Für \(n,k>0\) gilt

\[
s_{n,k}
=
s_{n-1,k-1}
+
(n-1)s_{n-1,k}.
\]

\vspace{0.3cm}

\textbf{Beweis:}

Sei \(N\) eine \(n\)-elementige Menge und \(a\in N\) fest gewählt.

Wir unterscheiden zwei Fälle:

\begin{itemize}
    \item \textbf{Fall 1:} \((a)\) ist ein eigener Zyklus.

    Dann bilden die restlichen \(n-1\) Elemente genau \(k-1\) Zyklen.

    \[
    s_{n-1,k-1}
    \]

    Möglichkeiten.

\end{itemize}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.12 (Fortsetzung)}

\begin{itemize}

    \item \textbf{Fall 2:} \(a\) liegt mit anderen Elementen in einem Zyklus.

    Dann bilden die restlichen \(n-1\) Elemente weiterhin \(k\) Zyklen.

    Dafür gibt es

    \[
    s_{n-1,k}
    \]

    Möglichkeiten.

    Im zweiten Fall kann \(a\) vor jedem der übrigen \(n-1\) Elemente eingefügt werden.

    Damit ergeben sich

    \[
    (n-1)s_{n-1,k}
    \]

    Möglichkeiten.

\end{itemize}

\vspace{0.3cm}

Addition der beiden Fälle ergibt \(s_{n,k}=s_{n-1,k-1}+(n-1)s_{n-1,k}\)

\hfill\(\square\)

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Definition 1.13 (Stirling-Zahlen zweiter Art)}

\vspace{0.3cm}

Die Stirling-Zahl zweiter Art

\[
S_{n,k}
\]

gibt die Anzahl der Partitionen einer
\(n\)-elementigen Menge in genau
\(k\) nichtleere Teilmengen an.

\vspace{0.4cm}

Es gelten

\[
S_{0,0}=1,
\qquad
S_{0,k}=0 \ (k>0),
\qquad
S_{n,0}=0 \ (n>0).
\]

\textbf{Beispiel}

\[
S_{3,2}=3.
\]

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Satz 1.14 (Stirling-Zahlen zweiter Art)}

\vspace{0.2cm}

Für \(n,k>0\) gilt

\[
S_{n,k}
=
S_{n-1,k-1}
+
kS_{n-1,k}.
\]

\vspace{0.3cm}

\textbf{Beweis:}

Sei \(N\) eine \(n\)-elementige Menge und \(a\in N\) fest gewählt.

Wir betrachten alle \(k\)-Partitionen von \(N\).

Wir unterscheiden zwei Fälle:

\begin{itemize}
    \item \textbf{Fall 1:} \(a\) bildet einen Einzelblock.

    Dann bleiben \(n-1\) Elemente übrig, die in \(k-1\) Blöcke zerlegt werden.

    \[
    S_{n-1,k-1}
    \]

    Möglichkeiten.

\end{itemize}

\end{frame}

\begin{frame}{1. Rekursionen}

\textbf{Beweis von Satz 1.14 (Fortsetzung)}

\begin{itemize}

    \item \textbf{Fall 2:} \(a\) liegt zusammen mit anderen Elementen in einem Block.

    Dann bleiben \(n-1\) Elemente übrig, die in \(k\) Blöcke zerlegt werden.

    \[
    S_{n-1,k}
    \]

    Möglichkeiten.

    Im zweiten Fall kann \(a\) zu einem der \(k\) Blöcke hinzugefügt werden.

    Damit ergeben sich

    \[
    kS_{n-1,k}
    \]

    Möglichkeiten.
    
\end{itemize}   

\vspace{0.3cm}

Die Addition der beiden Fälle ergibt \(S_{n,k}=S_{n-1,k-1}+kS_{n-1,k}\).

\hfill\(\square\)

\end{frame}

\begin{frame}{2. Diskrete Wahrscheinlichkeitsrechnung}

\textbf{Definition 2.1}

\vspace{0.2cm}

\textbf{a) Wahrscheinlichkeitsraum}

Ein Wahrscheinlichkeitsraum besteht aus einer endlichen Menge
\(\Omega\) und einer Abbildung

\[
p:\Omega\to[0,1]
\]

mit

\[
\sum_{\omega\in\Omega}p(\omega)=1.
\]

\vspace{0.2cm}

\textbf{b) Wahrscheinlichkeitsverteilung}

Die Abbildung \(p\) heißt Wahrscheinlichkeitsverteilung auf \(\Omega\).

\vspace{0.2cm}

\textbf{c) Ereignis}

Ein Ereignis ist eine Teilmenge \(A\subseteq\Omega\).

Für ein Ereignis gilt

\[
p(A)=\sum_{\omega\in A}p(\omega).
\]

\end{frame}

\begin{frame}{2. Diskrete Wahrscheinlichkeitsrechnung}

\textbf{Beispiel (Würfelwurf)}

\vspace{0.3cm}

Wir betrachten das Zufallsexperiment
\glqq Würfel werfen\grqq.

\vspace{0.3cm}

Der Ergebnisraum lautet

\[
\Omega=\{1,2,3,4,5,6\}.
\]

\vspace{0.4cm}

\textbf{Idealer Würfel}

\[
p(\omega)=\frac16
\qquad (\omega\in\Omega).
\]

\vspace{0.4cm}

\textbf{Gezinkter Würfel}

\[
p(1)=p(4)=\frac14,
\qquad
p(2)=p(5)=p(6)=\frac1{10},
\qquad
p(3)=\frac15.
\]

\end{frame}
